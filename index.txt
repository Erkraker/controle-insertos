function numeroBR(valor, casas = 2) {
return Number(valor).toLocaleString('pt-BR', { minimumFractionDigits: casas, maximumFractionDigits: casas });
}

function lerNumeros(texto) {
return texto.replace(/,/g, '.').split(/\s+|;/).map(v => v.trim()).filter(v => v !== '').map(Number).filter(v => !isNaN(v) && v > 0);
}

function normalizarNumeroOCR(valor) {
if (!valor) return null;
let texto = String(valor).trim();
texto = texto.replace(/[Oo]/g, '0');
texto = texto.replace(/,/g, '.');
const numero = Number(texto);
return isNaN(numero) ? null : numero;
}

function extrairDadosDoTextoOCR(texto) {
const limpo = texto
.replace(/[|]/g, ' ')
.replace(/[Oo]/g, '0')
.replace(/,/g, '.')
.replace(/\s+/g, ' ')
.trim();

const linhas = texto
.replace(/[Oo]/g, '0')
.replace(/,/g, '.')
.split(/\n+/)
.map(l => l.trim())
.filter(Boolean);

let quantidade = null;

// Prioridade: padrões com FURO / FUROS / QTD / QTDE / QUANTIDADE
const padroesQuantidade = [
/(?:QTD|QTDE|QUANTIDADE)\D{0,15}(\d{2,5})/i,
/(\d{2,5})\D{0,12}(?:FURO|FUROS)/i,
/(?:FURO|FUROS)\D{0,12}(\d{2,5})/i
];

for (const padrao of padroesQuantidade) {
const m = limpo.match(padrao);
if (m) {
quantidade = Number(m[1]);
break;
}
}

// Se não achou por texto, tenta número grande típico de quantidade em desenho.
if (!quantidade) {
const numsGrandes = [...limpo.matchAll(/\b\d{3,5}\b/g)]
.map(m => Number(m[0]))
.filter(n => n >= 100 && n <= 5000);
if (numsGrandes.length) {
quantidade = numsGrandes.sort((a, b) => b - a)[0];
}
}

// Extrai números candidatos para comprimento de furo.
// Faixa inicial pensada para comprimentos do desenho: 20 a 250 mm.
// Filtra diâmetro 16/18, códigos grandes, quantidade, ano, etc.
let candidatos = [...limpo.matchAll(/\b\d{1,3}(?:\.\d{1,3})?\b/g)]
.map(m => normalizarNumeroOCR(m[0]))
.filter(n => n !== null)
.filter(n => n >= 20 && n <= 250)
.filter(n => n !== 32)
.filter(n => n !== 16 && n !== 18)
.filter(n => !quantidade || n !== quantidade);

// Remove duplicados muito próximos, mantendo ordem.
let comprimentos = [];
for (const n of candidatos) {
if (!comprimentos.some(x => Math.abs(x - n) < 0.05)) {
comprimentos.push(n);
}
}

// Heurística para desenho de moenda: se aparecerem muitos números, prioriza os que têm cara de cota de comprimento.
// Mantém até 12 números para não encher o campo com lixo.
if (comprimentos.length > 12) {
comprimentos = comprimentos.filter(n => n >= 45 && n <= 120);
}
if (comprimentos.length > 12) {
comprimentos = comprimentos.slice(0, 12);
}

return { texto: limpo, comprimentos, quantidade };
}

async function processarImagemDesenho(event) {
const arquivo = event.target.files && event.target.files[0];
if (!arquivo) return;

const preview = document.getElementById('previewDesenho');
const status = document.getElementById('statusOCR');
const resultado = document.getElementById('resultadoOCR');
const textoBox = document.getElementById('textoOCRBox');

preview.src = URL.createObjectURL(arquivo);
preview.style.display = 'block';
status.innerHTML = 'Lendo desenho... isso pode demorar alguns segundos.';
resultado.classList.add('hidden');
textoBox.classList.add('hidden');

try {
if (typeof Tesseract === 'undefined') {
status.innerHTML = 'OCR não carregou. Verifique a internet e tente novamente.';
return;
}

const leitura = await Tesseract.recognize(arquivo, 'por+eng', {
logger: m => {
if (m.status === 'recognizing text') {
const pct = Math.round((m.progress || 0) * 100);
status.innerHTML = `Lendo texto do desenho... ${pct}%`;
}
}
});

const texto = leitura.data.text || '';
const dados = extrairDadosDoTextoOCR(texto);
ultimaLeituraOCR = dados;

document.getElementById('comprimentosEncontradosOCR').value = dados.comprimentos.map(n => String(n).replace('.', ',')).join('\n');
document.getElementById('quantidadeEncontradaOCR').value = dados.quantidade || '';
document.getElementById('textoOCR').textContent = texto.trim() || '(nenhum texto reconhecido)';

if (!dados.comprimentos.length && !dados.quantidade) {
status.innerHTML = 'Não consegui encontrar os dados principais. Tente foto mais nítida, mais reta e com melhor luz.';
} else {
status.innerHTML = `Leitura concluída. Confira antes de confirmar.`;
resultado.classList.remove('hidden');
}
} catch (erro) {
status.innerHTML = 'Erro ao ler a imagem. Tente novamente com outra foto.';
console.error(erro);
}
}

function melhorCombinacao(insertos, totalMetros, vidaInserto) {
const disponiveis = [...insertos].sort((a, b) => a - b);
const maxNovos = Math.ceil(totalMetros / vidaInserto) + 2;
let melhor = null;

function avaliar(combinacao) {
const soma = combinacao.reduce((a, b) => a + b, 0);
if (soma < totalMetros) return;
const sobra = soma - totalMetros;
const qtd = combinacao.length;
const menorSaldoUsado = Math.min(...combinacao);

if (!melhor || qtd < melhor.qtd || (qtd === melhor.qtd && sobra < melhor.sobra) || (qtd === melhor.qtd && sobra === melhor.sobra && menorSaldoUsado > melhor.menorSaldoUsado)) {
melhor = { combinacao, soma, sobra, qtd, menorSaldoUsado };
}
}

function gerarSubsets(index, atual) {
if (index === disponiveis.length) {
for (let novos = 0; novos <= maxNovos; novos++) {
const combinacao = [...atual];
for (let i = 0; i < novos; i++) combinacao.push(vidaInserto);
avaliar(combinacao);
}
return;
}
gerarSubsets(index + 1, atual);
gerarSubsets(index + 1, [...atual, disponiveis[index]]);
}

gerarSubsets(0, []);
if (!melhor) return [vidaInserto];
return melhor.combinacao.sort((a, b) => a - b);
}

function calcular(usarMelhorCombinacao) {
const nomePeca = document.getElementById('nomePeca').value.trim() || 'Peça sem nome';
const comprimentos = lerNumeros(document.getElementById('comprimentos').value);
const quantidadeFuros = Number(document.getElementById('quantidadeFuros').value);
const vidaInserto = Number(document.getElementById('vidaInserto').value) || 100;
let insertos = lerNumeros(document.getElementById('insertos').value);
const resultado = document.getElementById('resultado');

if (comprimentos.length === 0 || !quantidadeFuros || quantidadeFuros <= 0) {
resultado.innerHTML = '<div class="card">Preencha os comprimentos dos furos e a quantidade de furos.</div>';
return;
}

const mediaComprimento = comprimentos.reduce((a, b) => a + b, 0) / comprimentos.length;
const totalMetros = (mediaComprimento * quantidadeFuros) / 1000;
const grausPorMetro = 360 / totalMetros;

if (insertos.length === 0) insertos = [vidaInserto];

if (usarMelhorCombinacao) {
insertos = melhorCombinacao(insertos, totalMetros, vidaInserto);
document.getElementById('insertos').value = insertos.map(v => String(v).replace('.', ',')).join('\n');
}

let metrosRestantesPeca = totalMetros;
let grauAcumulado = 0;
let insertosUsados = [];
let indiceInserto = 0;

function usarInserto(saldoInserto, nomeInserto) {
const metrosUsados = Math.min(saldoInserto, metrosRestantesPeca);
const grausRodados = metrosUsados * grausPorMetro;
const grauInicio = grauAcumulado;
grauAcumulado += grausRodados;
metrosRestantesPeca -= metrosUsados;
const sobra = saldoInserto - metrosUsados;
insertosUsados.push({ nome: nomeInserto, saldoInicial: saldoInserto, grausRodados, grauInicio, grauFinal: grauAcumulado, metrosUsados, sobra });
}

while (metrosRestantesPeca > 0 && indiceInserto < insertos.length) {
usarInserto(insertos[indiceInserto], `Inserto ${indiceInserto + 1} (${numeroBR(insertos[indiceInserto])} m)`);
indiceInserto++;
}

while (metrosRestantesPeca > 0) {
usarInserto(vidaInserto, `Inserto ${insertosUsados.length + 1} novo`);
}

const ultimoInserto = insertosUsados[insertosUsados.length - 1];
const sobraFinal = ultimoInserto.sobra;
const mensagemTroca = insertosUsados.length === 1 ? `Não há troca. O inserto faz a peça toda, ou seja, 360°, e ainda sobra ${numeroBR(sobraFinal)} m para uma próxima peça.` : `Precisa de ${insertosUsados.length} insertos. A troca acontece conforme os graus abaixo.`;

const linhasTroca = insertosUsados.map((item, index) => {
const ultimo = index === insertosUsados.length - 1;
const completa = ultimo && item.grauFinal >= 359.999;
const textoFinal = completa ? 'completa a peça' : `até ${numeroBR(item.grauFinal)}°`;
return `<div class="troca-item">${item.nome} → ${textoFinal}</div>`;
}).join('');

const cardsInsertos = insertosUsados.map(item => `<div class="insert-card"><strong>${item.nome}</strong><br>Saldo inicial: ${numeroBR(item.saldoInicial)} m<br>Roda: ${numeroBR(item.grausRodados)}°<br>Trabalha de ${numeroBR(item.grauInicio)}° até ${numeroBR(item.grauFinal)}°<br>Consome: ${numeroBR(item.metrosUsados)} m<br>Sobra: ${numeroBR(item.sobra)} m</div>`).join('');

ultimoResultado = { tipo: 'calculo', data: new Date().toLocaleString('pt-BR'), nomePeca, mediaComprimento, totalMetros, grausPorMetro, quantidadeInsertos: insertosUsados.length, mensagemTroca, linhasTroca, insertosUsados, sobraFinal };

resultado.innerHTML = `<div class="result-box"><h2>${nomePeca}</h2><span class="badge">Furos: ${quantidadeFuros}</span><span class="badge">Média: ${numeroBR(mediaComprimento)} mm</span><span class="badge">Total: ${numeroBR(totalMetros)} m</span><span class="badge">${numeroBR(grausPorMetro, 4)}° por metro</span><div class="operator-report"><h3>Relatório do operador</h3><div class="answer-line"><strong>Quantos insertos preciso?</strong><br>${insertosUsados.length}</div><div class="answer-line"><strong>Trocas</strong>${linhasTroca}</div><div class="answer-line"><strong>Sobra</strong><br>${sobraFinal > 0 ? numeroBR(sobraFinal) + ' m para próxima peça' : 'Não sobra metragem'}</div></div><h3>Detalhamento técnico</h3><p><strong>Troca:</strong> ${mensagemTroca}</p>${cardsInsertos}<button onclick="salvarHistorico()">Salvar no histórico</button></div>`;
}

function definirBroca() {
const diametroFuro = Number(String(document.getElementById('diametroFuro').value).replace(',', '.'));
const profundidade = Number(String(document.getElementById('profundidadeFuro').value).replace(',', '.'));
const alturaFriso = Number(String(document.getElementById('alturaFriso').value).replace(',', '.')) || 0;
const deslocamentoX = Number(String(document.getElementById('deslocamentoX').value).replace(',', '.')) || 0;
const diametroCamisa = Number(String(document.getElementById('diametroCamisa').value).replace(',', '.')) || 0;
const resultadoBroca = document.getElementById('resultadoBroca');

if (!diametroFuro || !profundidade) {
resultadoBroca.innerHTML = '<div class="card">Preencha pelo menos o diâmetro do furo e a profundidade.</div>';
return;
}

const tabelaBrocas = {
16: { curta: 153, longa: 217 },
18: { curta: 172.5, longa: 244.5 }
};

const brocaDiametro = diametroFuro <= 16 ? 16 : 18;
const dados = tabelaBrocas[brocaDiametro];
const comprimentoNecessario = profundidade + alturaFriso + deslocamentoX;

let tipo = '';
let limite = 0;
let atende = true;

if (comprimentoNecessario <= dados.curta) {
tipo = 'Curta';
limite = dados.curta;
} else if (comprimentoNecessario <= dados.longa) {
tipo = 'Longa';
limite = dados.longa;
} else {
tipo = 'Não atende';
limite = dados.longa;
atende = false;
}

const folga = limite - comprimentoNecessario;
const status = atende ? 'OK para operação' : 'Não atende com as brocas cadastradas';

ultimoResultadoBroca = {
tipo: 'broca',
data: new Date().toLocaleString('pt-BR'),
diametroFuro,
profundidade,
alturaFriso,
deslocamentoX,
diametroCamisa,
brocaDiametro,
brocaTipo: tipo,
comprimentoNecessario,
limite,
folga,
status
};

resultadoBroca.innerHTML = `<div class="card ${atende ? 'broca-ok' : 'broca-alerta'}"><h2>Resultado da broca</h2><span class="badge">Furo Ø${numeroBR(diametroFuro)} mm</span><span class="badge">Broca Ø${brocaDiametro}</span><span class="badge">${tipo}</span><p><strong>Comprimento necessário:</strong> ${numeroBR(comprimentoNecessario)} mm</p><p><strong>Limite da broca:</strong> ${numeroBR(limite)} mm</p><p><strong>Folga:</strong> ${numeroBR(folga)} mm</p><p><strong>Diâmetro da camisa:</strong> ${diametroCamisa ? numeroBR(diametroCamisa) + ' mm' : '-'}</p><p><strong>Status:</strong> ${status}</p></div>`;
}

function salvarHistorico() {
if (!ultimoResultado) return;
const historico = getHistorico();
historico.unshift(ultimoResultado);
setHistorico(historico);
alert('Cálculo salvo no histórico.');
}

function salvarHistoricoBroca() {
if (!ultimoResultadoBroca) {
alert('Defina a broca antes de salvar.');
return;
}
const historico = getHistorico();
historico.unshift(ultimoResultadoBroca);
setHistorico(historico);
alert('Definição de broca salva no histórico.');
}

function carregarHistorico() {
const lista = document.getElementById('listaHistorico');
const historico = getHistorico();

if (historico.length === 0) { lista.innerHTML = '<div class="card">Nenhum cálculo salvo ainda.</div>'; return; }

lista.innerHTML = historico.map((item, index) => {
if (item.tipo === 'broca') {
return `<div class="history-item"><h3>Definição de Broca</h3><div class="small">${item.data}</div><span class="badge">Furo Ø${numeroBR(item.diametroFuro)} mm</span><span class="badge">Broca Ø${item.brocaDiametro} ${item.brocaTipo}</span><span class="badge">Necessário: ${numeroBR(item.comprimentoNecessario)} mm</span><p><strong>Status:</strong><br>${item.status}</p><div class="row-actions"><button class="danger" onclick="apagarItemHistorico(${index})">Apagar este</button></div></div>`;
}

return `<div class="history-item"><h3>${item.nomePeca}</h3><div class="small">${item.data}</div><span class="badge">Total: ${numeroBR(item.totalMetros)} m</span><span class="badge">Insertos: ${item.quantidadeInsertos}</span><span class="badge">Sobra: ${numeroBR(item.sobraFinal)} m</span><p><strong>Troca:</strong><br>${item.mensagemTroca}</p><div class="row-actions"><button class="danger" onclick="apagarItemHistorico(${index})">Apagar este</button></div></div>`;
}).join('');
}

function apagarItemHistorico(index) {
const historico = getHistorico();
historico.splice(index, 1);
setHistorico(historico);
carregarHistorico();
}

function limparHistorico() {
if (confirm('Deseja apagar todo o histórico?')) {
localStorage.removeItem(KEY_HISTORICO);
localStorage.removeItem('historicoInsertos');
localStorage.removeItem('historicoInsertos_v2');
document.getElementById('listaHistorico').innerHTML = '<div class="card">Nenhum cálculo salvo ainda.</div>';
}
}

function limparFormulario() {
document.getElementById('nomePeca').value = '';
document.getElementById('comprimentos').value = '';
document.getElementById('quantidadeFuros').value = '';
document.getElementById('vidaInserto').value = '100';
document.getElementById('insertos').value = '';
document.getElementById('resultado').innerHTML = '';
ultimoResultado = null;
}
